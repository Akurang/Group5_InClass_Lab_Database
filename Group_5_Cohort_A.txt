CREATE TRIGGER chk_booking_dates_update
BEFORE UPDATE ON Booking
FOR EACH ROW
BEGIN
    IF NEW.dateFrom <> OLD.dateFrom AND NEW.dateFrom <= CURDATE() THEN
        SIGNAL SQLSTATE '45000'
        SET MESSAGE_TEXT = 'Validation Error: New dateFrom must be in the future.';
    END IF;

    IF NEW.dateTo <> OLD.dateTo AND NEW.dateTo <= CURDATE() THEN
        SIGNAL SQLSTATE '45000'
        SET MESSAGE_TEXT = 'Validation Error: New dateTo must be in the future.';
    END IF;

    IF NEW.dateTo <= NEW.dateFrom THEN
        SIGNAL SQLSTATE '45000'
        SET MESSAGE_TEXT = 'Validation Error: dateTo must be greater than dateFrom.';
    END IF;
END$$
